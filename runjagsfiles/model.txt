model{
for (i in 1:m) {
# likelihood
y[i,1:2] ~ dmulti(prob[i,1:2], N[i])
prob[i,1] <- pi[i]*Se + (1-pi[i])*(1-Sp)
prob[i,2] <- pi[i]*(1-Se) + (1-pi[i])*Sp
# priors for prevalence parameters
pi[i] ~ dbeta(1,1)
}
Se~dbeta(HPSe[1], HPSe[2])
Sp~dbeta(HPSp[1], HPSp[2])
OR = (pi[2]/(1-pi[2])) / (pi[1]/(1-pi[1]))

}
